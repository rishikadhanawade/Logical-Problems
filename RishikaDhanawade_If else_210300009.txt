Q1
#################################################################################################
Accept the gross salary of an employee. Calculate and print the tax based on the given criteria : 
Salary                       Tax rate 
Up to Rs. 2,00,000           Nil
2,00,001 - 5,00,000          10%
Above 5,00,000               20%
#################################################################################################
 
#include<stdio.h>
int main()
{
    int grosssalary;
    float tax;
    printf("Enter the gross salary of employee");
    scanf("%d",&grosssalary);
    if(grosssalary<200001)
    {
        tax=0;
        printf("For grosssalary %d tax is %f",grosssalary,tax);
    }
    else if(grosssalary<=500000)
    {
        tax=0.1*grosssalary; //10% of salary
        printf("For grosssalary %d tax is %f",grosssalary,tax);
    }
    else
    {
        tax=0.2*grosssalary; //20% of salary
        printf("For grosssalary %d tax is %f",grosssalary,tax);
    }
}

Q2
####################################################################################################################################################################################################################################################################################
Accept the quantity of an item purchased and its price. Calculate the amount of purchase. If the amount exceeds Rs.5000, a discount of 20% is given otherwise the discount rate is 10%. Print the input values, the discount rate and amount, net amount to be paid by the customer.
####################################################################################################################################################################################################################################################################################
#include<stdio.h>
int main()
{
    int quantity,price;
    float discount;
    printf("Enter price per head and the quantity purchased");
    scanf("%d %d",&price,&quantity);
    printf("Enter price per head : %d \nThe quantity purchased: %d\n",price,quantity);
    int purchase_amount=quantity*price;
    printf("Total Purchase Amount: %d\n",purchase_amount);
    float net_amount_paid;
    if(purchase_amount>5000)
    {
        discount=0.2*purchase_amount;    //20% of Total Purchase Amount
        net_amount_paid=purchase_amount-discount;    //Net amount paid=Total Purchase Amount-Discount
        printf("Discount rate: %f \nNet amount paid: %f",discount,net_amount_paid);
    }
    else
    {
        discount=0.1*purchase_amount; //20% of Total Purchase Amount
        net_amount_paid=purchase_amount-discount; //Net amount paid=Total Purchase Amount-Discount
        printf("Discount rate: %f \nNet amount paid: %f",discount,net_amount_paid);
    }
}

Q4
########################################################################################################################
Accept the marks obtained by a student in 3 subjects out of 80 and print “ Very Well Done” if the percentage exceed 60%.
########################################################################################################################

#include<stdio.h>
int main()
{
    int marks1,marks2,marks3; 
    int total_marks_obtained;
    float percentage;
    int total_marks=80*3; //Each subject paper has maximum 80 marks
    printf("Total marks of all subkects: %d\n",total_marks);
    printf("Enter marks obtained by student in each of the three subjects");
    scanf("%d %d %d",&marks1,&marks2,&marks3);
    if((marks1>80)||(marks2>80)||(marks3>80))
    {
        printf("Invalid marks enter correct marks");
    }
    else
    {
        total_marks_obtained=marks1+marks2+marks3; //Calculate Total Marks Obtained by student
        printf("Total Marks Obtained by student: %d\n",total_marks_obtained);
        percentage=(1.0*total_marks_obtained/total_marks)*100.0;
        printf("Percentage: %f\n",percentage);
        if(percentage>60.000000f)
        printf("Very Well Done");
    }
}

#############################################################################################
Accept the age of a person and print an appropriate message as per the table given alongside: 
Age     Message 
<=12   You are a child 
13-19  You are a teenager 
20-59  You are an adult
 >=60  You are a senior citizen
#############################################################################################

#include<stdio.h>
int main()
{
    int age;
    printf("Enter your age: ");
    scanf("%d",&age);
    if(age>200) //lets assume maximum lifespan of a human is 200 years
    {
        printf("Invalid age");
    }
    else if(age<=12)
    printf("You are a child");
    else if (age>=13&&age<=19)
    printf("You are a teenager");
    else if (age>=20&&age<=59)
    printf("You are an adult");
    else
    printf("You are a senior citizen");
}

/* 1) Write a C program to calculate the factorial of a given number */

#include<stdio.h>
int main()
{
    int number,fact=1;
    printf("Enter a number");
    scanf("%d",&number);
    for(int i=1;i<=number;i++)  //loop to calculate factorial
    {
        fact=fact*i;
    }
    printf("Factorial is: %d",fact);
    return 0;
}

/* 3) Write a program in C to find the sum of the series 1 +11 + 111 + 1111 + .. n terms. */ 

#include<stdio.h>
int main()
{
    int i,n,count=0,sum=0;
    printf("Enter number of terms");
    scanf("%d",&n);
    for(i=1;count<n;i=i*10+1)
    {   if(count==n-1)
        {
            printf("%d ",i);
        }
        else
        printf("%d + ",i);
        sum=sum+i;
        count++;
    }
    printf("\nThe Sum is: %d",sum);
    return 0;
    }

/*  1
   12
   123
   1234 */

#include<stdio.h>
int main()
{
    int i,j,n=4,p;
    for(i=1;i<=n;i++)   //loop for rows
    {   p=1;
        for(j=1;j<=i;j++)  //loop for columns
        {
            printf("%d",p++);
        }
        printf("\n");  //for new line
    }
}

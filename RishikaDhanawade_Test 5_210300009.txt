/*
Q.1) Accept a number from the user and find the factorial of the number using recursion.
*/
#include<stdio.h>
int factorial(int n)
{
    if(n==0||n==1)
    return 1;
    else
    return (n*factorial(n-1));
}
int main()
{
   int number;
   printf("Enter a number");
   scanf("%d",&number);
   printf("Factorial is :%d",factorial(number));
   return 0;
}

/*
Q.2) Accept a sentence from the user and count number of words.
*/

#include<stdio.h>
int main()
{
   char str[100];
   printf("Enter a sentence: ");
   gets(str);
   int c=1;
   for(int i=0;str[i]!=0;i++)
   {
     if(str[i]==' '||str[i]=='\t'||str[i]=='\n')
     c++;
   }
   printf("number of words: %d",c);
   return 0;
}

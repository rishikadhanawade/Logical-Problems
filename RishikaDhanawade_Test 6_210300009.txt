// 1) Accept data in two 3*3 matrices and calculate the product of the matrices.

#include<stdio.h>
int main()
{
   int i,j,k,add;
   int a[3][3],b[3][3],c[3][3];
   printf("Enter the elements of the first matrix\n");
   for(i=0;i<3;i++)
   {
       for(j=0;j<3;j++)
       {
           scanf("%d",&a[i][j]);
       }
   }
   printf("\nEnter the elements of the first matrix\n");
    for(i=0;i<3;i++)
   {
       for(j=0;j<3;j++)
       {
           scanf("%d",&b[i][j]);
       }
   }
   for (i=0;i<3;i++) 
   {
      for (j=0;j<3;j++) 
      {
         add=0;
         for (k=0;k<3;k++) 
         {
            add = add + a[i][k]*b[k][j];
         }
         c[i][j] = add;
      }
   }
    printf("\nMultiplication Of Two Matrices : \n");
   for (i=0;i<3;i++) 
   {
      for (j=0;j<3;j++) 
      {
         printf(" %d ", c[i][j]);
      }
      printf("\n");
   }

   return 0;
}

// 2) Accept 10 numbers in an array; accept a number to be deleted from the array. Delete the number and print
// the new array

#include<stdio.h>
int main()
{
    int a[10],n,i,j,size=10;
    printf("Enter Array elements");
    for(i=0;i<size;i++)
    {
        scanf("%d",&a[i]);
    }
    printf("Accept number to be deleted");
    scanf("%d",&n);
    for(i=0;i<size;i++)
    {
       if(a[i]==n)
       {  
           printf("%d",a[i]);
           for(j=i+1;j<size;j++)
           {
               a[j-1]=a[j];
           }
           i--;
           size--;
       }
    }
   printf("New Array elements");
   for(i=0;i<size;i++)
    {
        printf("%d ",a[i]);
    }

}

Q.1 Print all special numbers between 1 to 200000
##################################################

#include<stdio.h>
int main()
{
    int num,fact=1,sum=0;
    int n;
    int d=0,i;
    for(num=1;num<=200000;num++)  //loop for covering numbers from 1 to 200000
    {
    for(n=num;n>0;n=n/10)     //loop for applying the special number criteria
    {
        d=n%10;
        for(i=d;i>0;i--)  //loop for finding out the factorial
        {
            fact=fact*i;
        }
        sum=sum+fact;
        fact=1; //resetting fact variable
    }
    if(sum==num)
    printf("%d ",num);
    d=0; //resetting d variable
    sum=0; //resetting sum variable
    }
}

#############################################
Q.2 Print all prime number between 51 to 100
#############################################

#include<stdio.h>
int main()
{
    int low=55,high=100,i;
    int count=0;
    int number;
    printf("Prime Nos between %d and %d are: ",low,high);
    for(number=low;number<=high;number++) //loop for covering numbers from 51 to 100
    {
        for(i=2;i<=number/2;i++) //loop for checking factors from 2 to number/2
        {
            if(number%i==0)
            {
                count++;
                break;
            }
        }
        if(count==0) //if number of factors between 2 to number/2 is zero it is a prime
        printf("%d ",number);
        count=0;
    }
    return 0;
}

Q.3 Accept a number from the user and check if it is a prime number or not
##########################################################################

#include<stdio.h>
int main()
{
int num,i,count=0;
printf("Enter a number");
scanf("%d",&num);
for(i=2;i<=num/2;i++) //loop for checking factors from 2 to number/2
{
    if(num%i==0)
    {
        count++;
        break;
    }
}
if(count==0&&num!=1) ///if number of factors between 2 to number/2 is zero it is a prime and even 1 is not prime
{
    printf("Yes it is a prime number");
}
else
printf("Not a prime number");
return 0;
}

##################################################
Q.4 Print all Armstrong number between 1 to 200000
##################################################

#include<stdio.h>
#include<math.h>
int main()
{
    int number,sum=0,digit=0,n,d=0;
    printf("Armstrong numbers between 1 and 200000 are: ");
    for(number=1;number<=200000;number++)        //loop for covering numbers from 1 to 200000
    {
        for(n=number;n>0;n=n/10)     //loop for counting the number of digits
         {
            digit++;          
         }
        for(n=number;n>0;n=n/10)         //loop for calculating the armstrong number formula
            {
            d=n%10;
            sum=sum+pow(d,digit);
            }
        if(sum==number)   //to check if it is an armstrong 
        printf("%d ",number); 
        digit=0;
        d=0;
        sum=0;
    }
}
